\section{Motivation and Background}

\begin{frame}{Motivation: LLMs reason, but they also calculate unreliably}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.54\textwidth}
\begin{block}{Observation behind \PAL{}}
Large language models can often translate a natural-language problem into useful intermediate steps, but may still make arithmetic or logical execution mistakes.
\end{block}
\vspace{0.15cm}
\begin{itemize}
  \item Few-shot prompting shows examples at test time.
  \item Chain-of-thought prompting asks the model to write intermediate reasoning steps.
  \item But in \CoT{}, the same model both \emph{plans} and \emph{executes} the solution.
\end{itemize}
\end{column}
\begin{column}{0.42\textwidth}
\begin{alertblock}{Problem}
Even if the decomposition is plausible, the final answer can be wrong because the LLM performs calculations as text generation rather than exact computation.
\end{alertblock}
\vspace{0.1cm}
\centering
\begin{tikzpicture}[node distance=0.28cm, every node/.style={font=\scriptsize}]
\node[draw,rounded corners,fill=blue!8,minimum width=3.6cm,align=center] (q) {Natural-language\ problem};
\node[draw,rounded corners,fill=blue!8,below=of q,minimum width=3.6cm,align=center] (steps) {Reasoning steps};
\node[draw,rounded corners,fill=red!10,below=of steps,minimum width=3.6cm,align=center] (calc) {LLM executes\ calculation};
\node[draw,rounded corners,fill=red!10,below=of calc,minimum width=3.6cm,align=center] (ans) {Final answer};
\draw[-{Latex}] (q) -- (steps);

\draw[-{Latex}] (calc) -- (ans);
\end{tikzpicture}
\end{column}
\end{columns}
\source{PAL paper, Abstract and Introduction.}
\end{frame}

% ---------------------------------------------------------
\begin{frame}{Background 1: few-shot prompting}
\begin{block}{Definition}
Few-shot prompting gives the LLM a small number of demonstrations directly in the input prompt. The model is not fine-tuned; it completes the new test instance by following the demonstrated pattern.
\end{block}
\vspace{0.2cm}
\[
  p = \langle x_1 \cdot y_1 \rangle \concat \langle x_2 \cdot y_2 \rangle \concat \dots \concat \langle x_k \cdot y_k \rangle
\]
\[
  \text{Inference input: } p \concat x_{\text{test}} \quad \Rightarrow \quad y_{\text{test}}
\]
\vspace{0.2cm}
\begin{exampleblock}{Intuition}
The prompt teaches the model the expected format and the task behavior from examples instead of changing model parameters.
\end{exampleblock}
\source{PAL paper, Section 2.}
\end{frame}

% ---------------------------------------------------------
\begin{frame}{Background 2: chain-of-thought prompting}
\begin{block}{Extension of few-shot prompting}
Chain-of-thought adds intermediate natural-language reasoning steps $t_i$ to each demonstration.
\end{block}
\[
  p = \langle x_1 \cdot t_1 \cdot y_1 \rangle \concat \langle x_2 \cdot t_2 \cdot y_2 \rangle \concat \dots \concat \langle x_k \cdot t_k \cdot y_k \rangle
\]
\[
  p \concat x_{\text{test}} \quad \Rightarrow \quad t_{\text{test}} \cdot y_{\text{test}}
\]
\vspace{0.15cm}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\textbf{Strength:} encourages decomposition into substeps.
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Weakness:} the LLM still has to execute every substep itself.
\end{column}
\end{columns}
\source{PAL paper, Section 2 and Figure 1.}
\end{frame}

% ---------------------------------------------------------
\begin{frame}[fragile]{Where chain-of-thought can fail}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\begin{block}{\CoT{} style}
\small
``The bakery baked 200 loaves. It sold 93 in the morning and 39 in the afternoon. 93 + 39 = 132. 6 were returned. So 200 - 132 - 6 = 62.''
\end{block}
\vspace{0.1cm}
\begin{alertblock}{Issue}
The verbal reasoning can look convincing, but a sign or arithmetic mistake changes the result.
\end{alertblock}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Correct computation}
\begin{lstlisting}[style=pythonstyle]
loaves_baked = 200
sold_morning = 93
sold_afternoon = 39
returned = 6
answer = loaves_baked - sold_morning \
         - sold_afternoon + returned
# answer = 74
\end{lstlisting}
\end{block}
\vspace{-0.1cm}
\begin{exampleblock}{Transition to \PAL{}}
Let the LLM write the computation, but let an interpreter run it.
\end{exampleblock}
\end{column}
\end{columns}
\source{Adapted from PAL paper, Figure 1.}
\end{frame}

\section{Program-aided Language Models}

\begin{frame}{Core idea of \PAL{}}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.50\textwidth}
\begin{block}{What changes?}
\PAL{} represents intermediate reasoning steps as a mixture of natural language and executable program statements.
\end{block}
\begin{itemize}
  \item LLM: parse the problem and generate code.
  \item Python interpreter: execute the generated program.
  \item Final answer: obtained from the runtime result.
\end{itemize}
\end{column}
\begin{column}{0.46\textwidth}
\centering
\begin{tikzpicture}[node distance=0.35cm, every node/.style={font=\scriptsize}]
\node[draw,rounded corners,fill=blue!8,minimum width=4.2cm,align=center] (q) {Natural-language problem};
\node[draw,rounded corners,fill=blue!8,below=of q,minimum width=4.2cm,align=center] (llm) {LLM generates programmatic steps};
\node[draw,rounded corners,fill=green!12,below=of llm,minimum width=4.2cm,align=center] (py) {Python interpreter executes};
\node[draw,rounded corners,fill=green!12,below=of py,minimum width=4.2cm,align=center] (a) {Answer};
\draw[-{Latex}] (q) -- (llm);
\draw[-{Latex}] (llm) -- (py);
\draw[-{Latex}] (py) -- (a);
\end{tikzpicture}
\end{column}
\end{columns}
\vspace{0.2cm}
\begin{alertblock}{Key conceptual shift}
The model no longer has to be a reliable calculator; it has to be a reliable translator from language to executable reasoning.
\end{alertblock}
\source{PAL paper, Section 3.}
\end{frame}

% ---------------------------------------------------------
\begin{frame}{Formal view of \PAL{} prompting}
\begin{block}{Prompt format}
In \PAL{}, each in-context example contains the question $x_i$ and a generated reasoning/program trace $t_i$, but the final answer does not need to be explicitly provided in the prompt because it will be obtained by execution.
\end{block}
\[
  p = \langle x_1 \cdot t_1 \rangle \concat \langle x_2 \cdot t_2 \rangle \concat \dots \concat \langle x_k \cdot t_k \rangle
\]
\[
  p \concat x_{\text{test}} \quad \Rightarrow \quad t_{\text{test}}
  \quad \xrightarrow{\text{Python}} \quad y_{\text{test}}
\]
\vspace{0.15cm}
\begin{exampleblock}{Important design choice}
The paper stresses meaningful variable names, because they help the LLM ground code variables in the entities from the question.
\end{exampleblock}
\source{PAL paper, Section 3 and Section 6.}
\end{frame}

% ---------------------------------------------------------
\begin{frame}[fragile]{Example: mathematical reasoning with \PAL{}}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.46\textwidth}
\begin{block}{Question}
Olivia has \$23. She bought five bagels for \$3 each. How much money does she have left?
\end{block}
\vspace{0.15cm}
\begin{itemize}
  \item The LLM identifies quantities and operations.
  \item The answer is produced by executing Python.
\end{itemize}
\end{column}
\begin{column}{0.50\textwidth}
\begin{lstlisting}[style=pythonstyle]
money_initial = 23
bagels = 5
bagel_cost = 3
money_spent = bagels * bagel_cost
money_left = money_initial - money_spent
answer = money_left
\end{lstlisting}
\end{column}
\end{columns}
\source{PAL paper, Figure 3.}
\end{frame}

% ---------------------------------------------------------
\begin{frame}[fragile]{Example: symbolic reasoning with \PAL{}}
\begin{block}{Task type: colored objects}
The model must track object order and attributes, then answer a relational question.
\end{block}
\vspace{0.1cm}
\begin{lstlisting}[style=pythonstyle]
objects = [
    ('paperclip', 'purple'),
    ('stress ball', 'pink'),
    ('keychain', 'brown'),
    ('scrunchiephone charger', 'green')
]

stress_ball_idx = None
for i, obj in enumerate(objects):
    if obj[0] == 'stress ball':
        stress_ball_idx = i
        break

direct_right = objects[stress_ball_idx + 1]
answer = direct_right[1]
\end{lstlisting}
\vspace{-0.1cm}
\source{Adapted from PAL paper, Figure 4.}
\end{frame}

% ---------------------------------------------------------
\begin{frame}{Analysis: what actually makes \PAL{} work?}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.49\textwidth}
\begin{block}{Interpreter matters}
When the model generated Python-like code but had to ``execute'' it itself, performance on GSM8K was far below full \PAL{}.
\end{block}
\begin{block}{Variable names matter}
Replacing meaningful variable names with random names reduced accuracy; names help ground program variables to problem entities.
\end{block}
\end{column}
\begin{column}{0.47\textwidth}
\begin{block}{Not restricted to code-only LMs}
The paper reports that sufficiently capable natural-language LMs can also benefit from \PAL{}, although weaker models may struggle with code generation.
\end{block}
\begin{block}{Runtime errors are rare}
The generated Python code was syntactically correct, and fewer than 1\% of examples raised runtime exceptions.
\end{block}
\end{column}
\end{columns}
\source{PAL paper, Section 6.}
\end{frame}

\section{Discussion and Conclusion}

\begin{frame}{Strengths and limitations}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\begin{exampleblock}{Strengths}
\begin{itemize}
  \item Clear neuro-symbolic division of labor.
  \item Strong gains on math, symbolic, and algorithmic tasks.
  \item Robustness to larger numbers.
  \item Simple implementation: prompt + interpreter.
\end{itemize}
\end{exampleblock}
\end{column}
\begin{column}{0.48\textwidth}
\begin{alertblock}{Limitations / questions}
\begin{itemize}
  \item Correctness still depends on generating the right program.
  \item Execution safety and sandboxing are not the central focus.
  \item Some tasks may not map cleanly to Python programs.
  \item Evaluation is based on few-shot prompting with specific model families.
\end{itemize}
\end{alertblock}
\end{column}
\end{columns}
\end{frame}

% ---------------------------------------------------------
\begin{frame}{Conclusion}
\begin{block}{Takeaway}
\PAL{} improves reasoning by making the LLM generate executable intermediate steps and delegating exact computation to a symbolic runtime.
\end{block}
\vspace{0.25cm}
\begin{itemize}
  \item It keeps the strength of LLMs: natural-language understanding and decomposition.
  \item It avoids a key weakness of LLMs: unreliable arithmetic and step execution.
  \item It provides a simple but powerful neuro-symbolic pattern: \textbf{language model as planner, interpreter as executor}.
\end{itemize}
\source{PAL paper, Conclusion.}
\end{frame}
