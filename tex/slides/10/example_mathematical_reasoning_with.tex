\begin{frame}[fragile]{Example of \(\langle \mathbf{x}_i \cdot \mathbf{t}_i \rangle\): mathematical reasoning with \PAL{}}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.46\textwidth}
\begin{block}{Question \(\mathbf{x}_i \downarrow \) \hfill Then \(\mathbf{t}_i \rightarrow\)}
Olivia has \$23. She bought five bagels for \$3 each. How much money does she have left? 
\end{block}
\end{column}
\begin{column}{0.50\textwidth}
\begin{lstlisting}[style=pythonstyle]
money_initial = 23
bagels = 5
bagel_cost = 3
money_spent = bagels * bagel_cost
money_left = money_initial - money_spent
answer = money_left
\end{lstlisting}
\end{column}
\end{columns}
\begin{alertblock}{Note}
  \begin{enumerate}
    \item Direct example results \(\mathbf{y}_i\) are not included in \(\mathbf{p}\) (\textit{see Code above}) and 
    \item the output \(\mathbf{y}\) should not be generated by the LLM eitherway.
  \end{enumerate}
  For the final output we have to perform this transition\[
    \mathbf{p} \concat \mathbf{x} \underset{\text{LLM}}{\rightarrow} \mathbf{t} \underset{\text{Python}}{\rightarrow} \mathbf{y}.
  \]
\end{alertblock}
